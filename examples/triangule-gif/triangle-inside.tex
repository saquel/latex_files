\begin{document}
	\foreach \i in {-30,-28,...,0}
	{		
	\begin{tikzpicture}[thick,violet]
		\node [draw, fill=violet!10, isosceles triangle, isosceles triangle apex angle=60, minimum size=-2*\i mm, rotate=\i,inner sep=0pt] at(0,0) {};
	\end{tikzpicture}
	}	
	
\end{document}/
